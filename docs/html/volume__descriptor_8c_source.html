21void
22create_volume_descriptor (volume_descriptor *vd, uint8_t type_code,
23 uint8_t version)
24{
25 vd->type_code = type_code;
26 strncpy (vd->identifier, "CD001", sizeof (vd->identifier));
27 vd->identifier[sizeof (vd->identifier) - 1] = '\0';
28 vd->version = version;
29}

80int8_t
81process_volume_descriptor_header (FILE *fptr, volume_descriptor *vd)
82{
83 uint8_t descriptor_type;
84 if (read_single_uint8 (fptr, &descriptor_type) != 0)
85 return HH_FREAD_ERROR;
86
87 fseek (fptr, 5, SEEK_CUR);
88
89 uint8_t descriptor_ver;
90 if (read_single_uint8 (fptr, &descriptor_ver) != 0)
91 return HH_FREAD_ERROR;
92
93 create_volume_descriptor (vd, descriptor_type, descriptor_ver);
94
95 return 0;
96}

98int8_t
99process_volume_descriptor_data (FILE *fptr, volume_descriptor_data *vdd)
100{
101 fseek (fptr, 1, SEEK_CUR); // Unused byte
102
103 if ((read_string (fptr, vdd->system_identifier, SYSTEM_IDENTIFIER_LEN) != 0)
104 || (read_string (fptr, vdd->volume_identifier, VOLUME_IDENTIFIER_LEN)
105 != 0))
106 {
107 return HH_FREAD_ERROR;
108 }
109
110 fseek (fptr, 8, SEEK_CUR); // Unused field
111
112 if (read_both_endian_data_uint32 (fptr, &vdd->volume_space_size) != 0)
113 return HH_FREAD_ERROR;
114
115 fseek (fptr, 32, SEEK_CUR); // Unused field
116
117 /* clang-format off */
118 if ((read_both_endian_data_uint16 (fptr, &vdd->volume_set_size) != 0)
119 || (read_both_endian_data_uint16 (fptr, &vdd->volume_sequence_number) != 0)
120 || (read_both_endian_data_uint16 (fptr, &vdd->logical_block_size) != 0))
121 {
122 return HH_FREAD_ERROR;
123 }
124 /* clang-format on */
125
126 if ((read_both_endian_data_uint32 (fptr, &vdd->path_table_size) != 0)
127 || (read_little_endian_data_uint32_t (fptr,
128 &vdd->type_l_path_table_location)
129 != 0)
130 || (read_little_endian_data_uint32_t (
131 fptr, &vdd->optional_type_l_path_table_location)
132 != 0)
133 || (read_little_endian_data_uint32_t (fptr,
134 &vdd->type_m_path_table_location)
135 != 0)
136 || (read_little_endian_data_uint32_t (
137 fptr, &vdd->optional_type_m_path_table_location)
138 != 0))
139 {
140 return HH_FREAD_ERROR;
141 }
142
143 if (read_array_uint8 (fptr, vdd->root_directory_entry,
144 ROOT_DIRECTORY_ENTRY_LEN)
145 != 0)
146 {
147 return HH_FREAD_ERROR;
148 }
149
150 /* clang-format off */
151 if ((read_string (fptr, vdd->volume_set_identifier, VOLUME_SET_IDENTIFIER_LEN) != 0)
152 || (read_string (fptr, vdd->publisher_identifier, PUBLISHER_IDENTIFIER_LEN) != 0)
153 || (read_string (fptr, vdd->data_preparer_identifier, DATA_PREPARER_IDENTIFIER_LEN) != 0)
154 || (read_string (fptr, vdd->application_identifier, APPLICATION_IDENTIFIER_LEN) != 0)
155 || (read_string (fptr, vdd->copyright_file_identifier, COPYRIGHT_FILE_IDENTIFIER_LEN) != 0)
156 || (read_string (fptr, vdd->abstract_file_identifier, ABSTRACT_FILE_IDENTIFIER_LEN) != 0)
157 || (read_string (fptr, vdd->bibliographic_file_identifier, BIBLIOGRAPHIC_FILE_IDENTIFIER_LEN) != 0))
158 {
159 return HH_FREAD_ERROR;
160 }
161 /* clang-format on */
162
163 if ((read_dec_datetime (fptr, &vdd->volume_creation_date_and_time) != 0)
164 || (read_dec_datetime (fptr, &vdd->volume_modification_date_and_time)
165 != 0)
166 || (read_dec_datetime (fptr, &vdd->volume_expiration_date_and_time) != 0)
167 || (read_dec_datetime (fptr, &vdd->volume_effective_date_and_time) != 0))
168 {
169 return HH_FREAD_ERROR;
170 }
171
172 if (read_single_uint8 (fptr, &vdd->file_structure_version) != 0)
173 return HH_FREAD_ERROR;
174
175 fseek (fptr, 1, SEEK_CUR); // Unused byte
176
177 read_string (fptr, vdd->DAT_file_creation_software_identifier,
178 DAT_FILE_CREATION_SOFTWARE_IDENTIFIER_LEN);
179
180 fseek (fptr, 1, SEEK_CUR); // Unused byte
181
182 read_string (fptr, vdd->DAT_file_creation_software_version_number,
183 DAT_FILE_CREATION_SOFTWARE_VERSION_NUMBER_LEN);
184
185 fseek (fptr, 1, SEEK_CUR); // Unused byte
186
187 fseek (fptr, 0x1EC, SEEK_CUR); // Unknown data (492 bytes)
188
189 fseek (fptr, 0x28D, SEEK_CUR); // Reserved by ISO
190
191 return 0;
192}
