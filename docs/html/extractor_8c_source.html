16 /*
17 * the `file_identifier` terminates with a `;` character followed by the
18 * file ID number in ASCII coded decimal (`1`).
19 * See: https://wiki.osdev.org/ISO_9660#Directories
20 */
21 char *actual_filename = strtok (dr->file_identifier, (const char *)";");
22 if (actual_filename == NULL)
23 {
24 /*
25 * Just use the default/existing filename.
26 * It'll be incorrect, but probably not worth stoping execution over.
27 * Users could just manually remove the `;1` part; the data itself
28 * should be fine.
29 */
30 actual_filename = dr->file_identifier;
31 }
32
33 printf ("Extracting file: %s\n", actual_filename);
34
35 // +1 for the null terminator, +1 for `/` between dir and filename
36 size_t filename_length = strlen (path) + strlen (actual_filename) + 2;
37
38 char *output_filename = (char *)calloc (filename_length, sizeof (char));
39 if (output_filename == NULL)
40 {
41 perror ("ERROR: unable to calloc `output_filename`.");
42 return -1;
43 }
44
45 strcpy (output_filename, path);
46 strcat (output_filename, &OP_PATH_SEPARATOR);
47 strcat (output_filename, actual_filename);
48
49 FILE *output_file = fopen (output_filename, "wb");
50 if (output_file == NULL)
51 {
52 perror ("ERROR: unable to open output file.");
53 free (output_filename);
54 return -1;
55 }
56
57 // `j` must be in hex, otherwise `data_length` can be treated as an int value
58 for (uint32_t j = 0x0; j < dr->data_length; j++)
59 {
60 uint8_t byte = read_single_uint8 (fptr);
61 fwrite (&byte, sizeof (uint8_t), 1, output_file);
62 }
63
64 fclose (output_file);

88 {
89 directory_record curr_file = dir.records[i];
90
91 if (curr_file.file_flags.subdirectory)

93 continue;
