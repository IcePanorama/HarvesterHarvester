25int8_t
26extract_file (FILE *fptr, directory_record *dr, const char *path)
27{
33 char *actual_filename = strtok (dr->file_identifier, (const char *)";");
34 if (actual_filename == NULL)
35 {
42 actual_filename = dr->file_identifier;
43 }
44
45 printf ("Extracting file: %s\n", actual_filename);
46
47 // +1 for the null terminator, +1 for `/` between dir and filename
48 size_t filename_length = strlen (path) + strlen (actual_filename) + 2;
49
50 char *output_filename = (char *)calloc (filename_length, sizeof (char));
51 if (output_filename == NULL)
52 {
53 fprintf (stderr, CALLOC_FAILED_ERR_MSG_FMT, filename_length);
54 return HH_MEM_ALLOC_ERROR;
55 }
56
57 strcpy (output_filename, path);
58 strcat (output_filename, &OP_PATH_SEPARATOR);
59 strcat (output_filename, actual_filename);
60
61 FILE *output_file = fopen (output_filename, "wb");
62 if (output_file == NULL)
63 {
64 fprintf (stderr, FOPEN_FAILED_ERR_MSG_FMT, output_filename);
65 free (output_filename);
66 return HH_FOPEN_ERROR;
67 }
68
69 // `j` must be in hex, otherwise `data_length` can be treated as an int value
70 for (uint32_t j = 0x0; j < dr->data_length; j++)
71 {
72 uint8_t byte;
73 if (read_single_uint8 (fptr, &byte) != 0)
74 {
75 fclose (output_file);
76 free (output_filename);
77 return HH_FREAD_ERROR;
78 }
79
80 fwrite (&byte, sizeof (uint8_t), 1, output_file);
81 }
82
83 fclose (output_file);
84 free (output_filename);
85 return 0;
86}

88int8_t
89extract_directory (FILE *fptr, const uint16_t block_size, const char *path)
90{
91 directory dir;
92 create_directory (&dir);
93 process_directory (fptr, &dir);
94
95 printf ("Extracting directory: %s\n", path);
96
97 for (size_t i = 0x0; i < dir.current_record; i++)
98 {
99 directory_record curr_file = dir.records[i];
100
101 if (curr_file.file_flags.subdirectory)
102 {
103 continue;
104 }
105 else if (OP_DEBUG_MODE
106 && curr_file.data_length > OP_DEBUG_FILE_SIZE_LIMIT)
107 {
108 printf ("[DEBUG_MODE] Skipping file, %s.\n",
109 curr_file.file_identifier);
110 continue;
111 }
112
113 fseek (fptr, curr_file.location_of_extent * block_size, SEEK_SET);
114
115 if (extract_file (fptr, &curr_file, path) != 0)
116 {
117 destroy_directory (&dir);
118 return -1;
119 }
120 }
121
122 destroy_directory (&dir);
123 return 0;
124}

126int8_t
127create_directories_and_extract_data_from_path_file (FILE *fptr,
128 uint16_t block_size,
129 path_table *pt)
130{
131 for (size_t i = pt->current_entry - 1; i > 0; --i)
132 {
133 path_table_entry curr_dir = pt->entries[i];
134 path_table_entry target_dir = curr_dir;
135
136 // supports 10 levels of directories which is probably overkill.
137 const uint32_t PATH_MAX_LEN
138 = ((curr_dir.directory_identifier_length + 1) * 10)
139 + (strlen (OP_OUTPUT_DIR) + 1) + (strlen (CURRENT_DISK_NAME) + 1)
140 + 1;
141 char *path = calloc (PATH_MAX_LEN, sizeof (char));
142 if (path == NULL)
143 {
144 fprintf (stderr, CALLOC_FAILED_ERR_MSG_FMT, PATH_MAX_LEN);
145 return HH_MEM_ALLOC_ERROR;
146 }
147
148 strcat (path, curr_dir.directory_identifier);
149
150 do
151 {
152 uint16_t index
153 = change_endianness_uint16 (curr_dir.parent_directory_number);
154
155 // parent_directory_number is 1-based
156 curr_dir = pt->entries[index - 1];
157
158 if (prepend_path_string (path,
159 (const char *)curr_dir.directory_identifier)
160 != 0)
161 {
162 free (path);
163 return -1;
164 }
165 }
166 while (curr_dir.parent_directory_number > 0x0100);
167
168 if (create_output_directory (path) != 0)
169 return -1;
170
171 fseek (fptr, block_size * target_dir.location_of_extent, SEEK_SET);
172
173 if (extract_directory (fptr, block_size, path) != 0)
174 return -1;
175
176 free (path);
177 }
178
179 return 0;
180}
