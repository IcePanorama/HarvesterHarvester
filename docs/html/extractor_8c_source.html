26int8_t
27extract_file_using_dir_record (FILE *fptr, struct directory_record *dr,
28 const char *path)
29{
35 char *actual_filename = strtok (dr->file_identifier, (const char *)";");
36 if (actual_filename == NULL)
37 {
44 actual_filename = dr->file_identifier;
45 }
46
47 hh_log (HH_LOG_INFO, "Extracting file: %s", actual_filename);
48
49 // +1 for the null terminator, +1 for `/` between dir and filename
50 size_t filename_length = strlen (path) + strlen (actual_filename) + 2;
51
52 char *output_filename = (char *)calloc (filename_length, sizeof (char));
53 if (output_filename == NULL)
54 {
55 return handle_calloc_error (filename_length);
56 }
57
58 strcpy (output_filename, path);
59 strcat (output_filename, &OP_PATH_SEPARATOR);
60 strcat (output_filename, actual_filename);
61
62 FILE *output_file = fopen (output_filename, "wb");
63 if (output_file == NULL)
64 {
65 handle_fopen_error (output_filename);
66 free (output_filename);
67 return HH_FOPEN_ERROR;
68 }
69
70 // `j` must be in hex, otherwise `data_length` can be treated as an int value
71 for (uint32_t j = 0x0; j < dr->data_length; j++)
72 {
73 uint8_t byte;
74 if (read_single_uint8 (fptr, &byte) != 0)
75 {
76 fclose (output_file);
77 free (output_filename);
78 return HH_FREAD_ERROR;
79 }
80
81 fwrite (&byte, sizeof (uint8_t), 1, output_file);
82 }
83
84 fclose (output_file);
85 free (output_filename);
86 return 0;
87}

89int8_t
90extract_directory (FILE *fptr, const uint16_t block_size, const char *path)
91{
92 directory dir;
93 create_directory (&dir);
94 process_directory (fptr, &dir);
95
96 hh_log (HH_LOG_INFO, "Extracting directory: %s", path);
97
98 for (size_t i = 0x0; i < dir.current_record; i++)
99 {
100 directory_record curr_file = dir.records[i];
101
102 if (curr_file.file_flags.subdirectory)
103 {
104 continue;
105 }
106 else if (OP_DEBUG_MODE
107 && curr_file.data_length > OP_DEBUG_FILE_SIZE_LIMIT)
108 {
109 hh_log (HH_LOG_INFO, "[DEBUG_MODE]Skipping file, %s.",
110 curr_file.file_identifier);
111 continue;
112 }
113
114 fseek (fptr, curr_file.location_of_extent * block_size, SEEK_SET);
115
116 if (extract_file_using_dir_record (fptr, &curr_file, path) != 0)
117 {
118 destroy_directory (&dir);
119 return -1;
120 }
121 }
122
123 destroy_directory (&dir);
124 return 0;
125}

127int8_t
128create_directories_and_extract_data_from_path_file (FILE *fptr,
129 uint16_t block_size,
130 path_table *pt)
131{
132 for (size_t i = pt->current_entry - 1; i > 0; --i)
133 {
134 path_table_entry curr_dir = pt->entries[i];
135 path_table_entry target_dir = curr_dir;
136
137 // supports 10 levels of directories which is probably overkill.
138 const uint32_t PATH_MAX_LEN
139 = ((curr_dir.directory_identifier_length + 1) * 10)
140 + (strlen (OP_OUTPUT_DIR) + 1) + (strlen (CURRENT_DISK_NAME) + 1)
141 + 1;
142 char *path = calloc (PATH_MAX_LEN, sizeof (char));
143 if (path == NULL)
144 {
145 return handle_calloc_error (PATH_MAX_LEN);
146 }
147
148 strcat (path, curr_dir.directory_identifier);
149
150 do
151 {
152 uint16_t index
153 = change_endianness_uint16 (curr_dir.parent_directory_number);
154
155 // parent_directory_number is 1-based
156 curr_dir = pt->entries[index - 1];
157
158 if (prepend_path_string (path,
159 (const char *)curr_dir.directory_identifier)
160 != 0)
161 {
162 free (path);
163 return -1;
164 }
165 }
166 while (curr_dir.parent_directory_number > 0x0100);
167
168 if (create_output_directory (path) != 0)
169 return -1;
170
171 fseek (fptr, block_size * target_dir.location_of_extent, SEEK_SET);
172
173 if (extract_directory (fptr, block_size, path) != 0)
174 return -1;
175
176 free (path);
177 }
178
179 return 0;
180}

182int8_t
183extract_file_using_idx_entry (FILE *fptr, index_entry *idx, const char *path)
184{
185 hh_log (HH_LOG_INFO, "Extracting file: %s", path);
186
187 FILE *output_file = fopen (path, "wb");
188 if (output_file == NULL)
189 {
190 char tmp[256] = { 0 };
191 strncpy (tmp, path, strlen (path) - strlen (idx->filename));
192
193 if (create_new_output_directory (tmp) != 0)
194 return -1;
195
196 output_file = fopen (path, "wb");
197 if (output_file == NULL)
198 {
199 hh_log (HH_LOG_ERROR, "Error opening output file, %s.", path);
200 return HH_FOPEN_ERROR;
201 }
202 }
203
204 fseek (fptr, idx->start, SEEK_SET);
205
206 for (uint32_t i = 0x0; i < idx->size; i++)
207 {
208 uint8_t byte;
209 if (read_single_uint8 (fptr, &byte) != 0)
210 {
211 fclose (output_file);
212 return -1;
213 }
214
215 fwrite (&byte, sizeof (uint8_t), 1, output_file);
216 }
217
218 fclose (output_file);
219 return 0;
220}

222int8_t
223extract_index_file (index_file *idx, const char *idx_path,
224 const char *dat_path)
225{
226 FILE *dat_file = fopen (dat_path, "rb");
227 if (dat_file == NULL)
228 {
229 return handle_fopen_error ((char *)dat_path);
230 }
231
232 for (size_t i = 0; i < idx->current_index; i++)
233 {
234 char output_path[256] = { 0 };
235
240 strncpy (output_path, idx_path, strlen (idx_path) - 10);
241 strcat (output_path, idx->indicies[i].full_path);
242
243 if (extract_file_using_idx_entry (dat_file, &idx->indicies[i],
244 output_path)
245 != 0)
246 {
247 fclose (dat_file);
248 return -1;
249 }
250 }
251
252 fclose (dat_file);
253 return 0;
254}
