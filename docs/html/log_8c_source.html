28void
29print_hex_data (unsigned char *buffer, const uint8_t BUFFER_LEN)
30{
31 for (uint8_t i = 0; i < BUFFER_LEN; i++)
32 {
33 printf (" %02x", buffer[i]);
34 if ((i != 0) && (i % 4 == 3))
35 {
36 printf ("\t");
37 }
38 if ((i != 0) && (i % 16 == 15))
39 {
40 puts ("");
41 }
42 }
43 puts ("");
44}

52void
53print_some_data_from_file (FILE *fptr)
54{
55 unsigned char buffer[BYTES_TO_READ];
56 size_t bytes_read = fread (buffer, sizeof (buffer[0]), BYTES_TO_READ, fptr);
57 if (bytes_read != BYTES_TO_READ)
58 {
59 handle_fread_error (bytes_read, BYTES_TO_READ);
60 return;
61 }
62
63 print_hex_data (buffer, BYTES_TO_READ);
64 fseek (fptr, -BYTES_TO_READ, SEEK_CUR);
65}
