33int8_t
34create_index_file (index_file *i)
35{
36 i->indicies = malloc (IDX_STARTING_NUM_INDICIES * sizeof (index_entry));
37 if (i->indicies == NULL)
38 {
39 hh_log (HH_LOG_ERROR, "Unable to malloc indicies of size %zu.",
40 IDX_STARTING_NUM_INDICIES * sizeof (index_entry));
41 return -1;
42 }
43
44 i->size = IDX_STARTING_NUM_INDICIES;
45 i->current_index = 0;
46
47 return 0;
48}

57int8_t
58process_index_file (FILE *fptr, index_file *idxf)
59{
60 while (!peek_eof (fptr))
61 {
62 index_entry entry;
63 entry.entry_start = ftell (fptr);
64
65 fseek (fptr, 0x5, SEEK_CUR); // skip `XFLE#`
66
67 if (!peek_char_is (fptr, ':'))
68 {
69 hh_log (HH_LOG_ERROR,
70 "Unexpected character in index file. Aborting processing.");
71 return -1;
72 }
73
74 if (read_string (fptr, entry.full_path, (uint8_t)FULL_PATH_MAX_LEN) != 0)
75 return HH_FREAD_ERROR;
76
77 uint8_t len = strlen (entry.full_path);
78 char *last_word = entry.full_path;
79 for (uint8_t i = 0; i < len; i++)
80 {
81 if (entry.full_path[i] == '\\')
82 {
83 last_word = entry.full_path + i + 1;
84 entry.full_path[i] = OP_PATH_SEPARATOR;
85 }
86 }
87 strcpy (entry.filename, last_word);
88
95 fseek (fptr, entry.entry_start + (0x94 - 0x10), SEEK_SET);
96
97 /* clang-format off */
98 if ((read_little_endian_data_uint32_t (fptr, &entry.start) != 0)
99 || (read_little_endian_data_uint32_t (fptr, &entry.size) != 0))
100 {
101 return HH_FREAD_ERROR;
102 }
103 /* clang-format on */
104
105 fseek (fptr, 0x4, SEEK_CUR); // Skip zeros
106
107 // Very unnecessary, but we might as well double check our data
108 uint32_t value;
109 if (read_little_endian_data_uint32_t (fptr, &value) != 0)
110 return HH_FREAD_ERROR;
111
112 if (entry.size != value)
113 {
114 hh_log (HH_LOG_ERROR, "Expected 0x%08X, got 0x%08X.", entry.size,
115 value);
116 return -1;
117 }
118
119 if (add_entry_to_index_file (idxf, &entry) != 0)
120 {
121 return -1;
122 }
123 }
124
125 return 0;
126}

138int8_t
139add_entry_to_index_file (index_file *file, index_entry *entry)
140{
141 if (file->current_index >= file->size)
142 {
143 if (resize_indicies (file) != 0)
144 {
145 return -1;
146 }
147 }
148
149 file->indicies[file->current_index] = *entry;
150 file->current_index++;
151 return 0;
152}

154int8_t
155resize_indicies (index_file *idx)
156{
157 size_t new_size = idx->size * IDX_RECORDS_GROWTH_RATE;
158 index_entry *new_indicies = (index_entry *)realloc (
159 idx->indicies, new_size * sizeof (index_entry));
160 if (new_indicies == NULL)
161 {
162 destroy_index_file (idx);
163 return handle_realloc_error ("indicies", idx->size, new_size);
164 }
165
166 idx->indicies = new_indicies;
167 idx->size = new_size;
168 return 0;
169}
