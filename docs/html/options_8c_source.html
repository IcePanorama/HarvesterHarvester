43void
44handle_command_line_args (int argc, char **argv)
45{
46 int i;
47 for (i = 1; i < argc; i++)
48 {
49 if (strncmp (argv[i], "-d", 2) == 0)
50 {
51 i++;
52 size_t path_len = strlen (argv[i]);
53 if (argv[i][path_len - 1] == '/' || argv[i][path_len - 1] == '\\')
54 {
55 argv[i][path_len - 1] = '\0';
56 }
57 OP_INPUT_DIR = argv[i];
58 }
59 else if (strncmp (argv[i], "-o", 2) == 0)
60 {
61 i++;
62 size_t path_len = strlen (argv[i]);
63 if (argv[i][path_len - 1] == '/' || argv[i][path_len - 1] == '\\')
64 {
65 argv[i][path_len - 1] = '\0';
66 }
67 OP_OUTPUT_DIR = argv[i];
68 }
69 else if (argv[i][0] == '-')
70 {
71 process_flag (argv[i]);
72 }
73 else
74 {
75 break;
76 }
77 }
78
79 if (i == argc)
80 return;
81
82 if (is_string_dat_file (argv[argc - 1]))
83 {
84 // just processing the user-provided file @ the user-provided path
85 OP_BATCH_PROCESS = false;
86 }
87 else
88 {
89 handle_unknown_command_line_argument_error (argv[argc - 1]);
90 }
91}

121void
122process_flag (char *f)
123{
124 if (strcmp (f, "--debug") == 0)
125 {
126 OP_DEBUG_MODE = true;
127 }
128 else if (strcmp (f, "--help") == 0)
129 {
130 print_out_list_of_command_line_arguments ();
131 exit (0);
132 }
133 else if (strcmp (f, "-e") == 0)
134 {
135 OP_SKIP_INT_DAT_PROCESSING = true;
136 }
137 else if (strcmp (f, "-i") == 0)
138 {
139 OP_SKIP_DAT_PROCESSING = true;
140 }
141 else if (strcmp (f, "--version") == 0)
142 {
143 printf ("HarvesterHarvester v%s\n", VERSION_NUMBER_STR);
144 exit (0);
145 }
146 else
147 {
148 handle_unknown_command_line_argument_error (f);
149 }
150}
