36int8_t
37read_both_endian_data_uint32 (FILE *fptr, uint32_t *output)
38{
39 uint32_t value;
40 if (read_little_endian_data_uint32_t (fptr, &value) != 0)
41 return HH_FREAD_ERROR;
42
43 uint32_t expected_value;
44 read_big_endian_data_uint32_t (fptr, &expected_value);
45
46 if (value != expected_value)
47 {
48 hh_log (HH_LOG_ERROR,
49 "ERROR: Incorrect endian conversion (uint32_t). Expected "
50 "0x%08X, got 0x%08X.",
51 expected_value, value);
52 }
53
54 *output = value;
55 return 0;
56}

80int8_t
81read_little_endian_data_uint32_t (FILE *fptr, uint32_t *output)
82{
83 uint8_t bytes[4];
84 size_t bytes_read = fread (bytes, sizeof (uint8_t), 4, fptr);
85 if (bytes_read != 4)
86 {
87 handle_fread_error (bytes_read, sizeof (bytes));
88
89 return HH_FREAD_ERROR;
90 }
91
92 *output = ((uint32_t)bytes[3] << 24) | ((uint32_t)bytes[2] << 16)
93 | ((uint32_t)bytes[1] << 8) | (uint32_t)bytes[0];
94
95 return 0;
96}

144int8_t
145read_string (FILE *fptr, char *output, uint8_t length)
146{
147 size_t bytes_read = fread (output, sizeof (char), length - 1, fptr);
148 output[length - 1] = '\0';
149 if (bytes_read != sizeof (char) * length - 1)
150 {
151 handle_fread_error (bytes_read, sizeof (char) * length - 1);
152 return HH_FREAD_ERROR;
153 }
154
155 return 0;
156}

171int8_t
172read_dec_datetime (FILE *fptr, dec_datetime *dt)
173{
174 if ((read_string (fptr, dt->year, YEAR_FIELD_LEN) != 0)
175 || (read_string (fptr, dt->month, MONTH_FIELD_LEN) != 0)
176 || (read_string (fptr, dt->day, DAY_FIELD_LEN) != 0)
177 || (read_string (fptr, dt->hour, HOUR_FIELD_LEN) != 0)
178 || (read_string (fptr, dt->minute, MINUTE_FIELD_LEN) != 0)
179 || (read_string (fptr, dt->second, SECOND_FIELD_LEN) != 0)
180 || (read_string (fptr, dt->hundredths_of_a_second,
181 HUNDREDTHS_OF_A_SECOND_FIELD_LEN)
182 != 0)
183 || (read_single_uint8 (fptr, &dt->time_zone_offset) != 0))
184 {
185 return HH_FREAD_ERROR;
186 }
187
188 return 0;
189}

221int8_t
222read_file_flags (FILE *fptr, file_flags *ff)
223{
224 uint8_t byte;
225 if (read_single_uint8 (fptr, &byte) != 0)
226 return HH_FREAD_ERROR;
227
228 if (byte & 0x1)
229 {
230 ff->hidden = true;
231 }
232 if (byte & 0x2)
233 {
234 ff->subdirectory = true;
235 }
236 if (byte & 0x4)
237 {
238 ff->associated_file = true;
239 }
240 if (byte & 0x8)
241 {
242 ff->extended_attribute_record_contains_owner_and_group_permissions
243 = true;
244 }
245 if (byte & 0x10)
246 {
247 ff->final_directory_record = true;
248 }
249
250 return 0;
251}
