35int8_t
36read_both_endian_data_uint32 (FILE *fptr, uint32_t *output)
37{
38 uint32_t value;
39 if (read_little_endian_data_uint32_t (fptr, &value) != 0)
40 return HH_FREAD_ERROR;
41
42 uint32_t expected_value;
43 read_big_endian_data_uint32_t (fptr, &expected_value);
44
45 if (value != expected_value)
46 {
47 printf ("ERROR: Incorrect endian conversion (uint32_t).\n\tExpected "
48 "%08X, got %08X.\n",
49 expected_value, value);
50 }
51
52 *output = value;
53 return 0;
54}

77int8_t
78read_little_endian_data_uint32_t (FILE *fptr, uint32_t *output)
79{
80 uint8_t bytes[4];
81 size_t bytes_read = fread (bytes, sizeof (uint8_t), 4, fptr);
82 if (bytes_read != 4)
83 {
84 handle_fread_error (bytes_read, sizeof (bytes));
85
86 return HH_FREAD_ERROR;
87 }
88
89 *output = ((uint32_t)bytes[3] << 24) | ((uint32_t)bytes[2] << 16)
90 | ((uint32_t)bytes[1] << 8) | (uint32_t)bytes[0];
91
92 return 0;
93}

142read_string (FILE *fptr, char *output, uint8_t length)
143{
144 size_t bytes_read = fread (output, sizeof (char), length - 1, fptr);
145 output[length - 1] = '\0';
146 if (bytes_read != sizeof (char) * length - 1)
147 {
148 handle_fread_error (bytes_read, sizeof (char) * length - 1);
149 return HH_FREAD_ERROR;

168int8_t
169read_dec_datetime (FILE *fptr, dec_datetime *dt)
170{
171 if ((read_string (fptr, dt->year, YEAR_FIELD_LEN) != 0)
172 || (read_string (fptr, dt->month, MONTH_FIELD_LEN) != 0)
173 || (read_string (fptr, dt->day, DAY_FIELD_LEN) != 0)
174 || (read_string (fptr, dt->hour, HOUR_FIELD_LEN) != 0)
175 || (read_string (fptr, dt->minute, MINUTE_FIELD_LEN) != 0)
176 || (read_string (fptr, dt->second, SECOND_FIELD_LEN) != 0)
177 || (read_string (fptr, dt->hundredths_of_a_second,
178 HUNDREDTHS_OF_A_SECOND_FIELD_LEN)
179 != 0)
180 || (read_single_uint8 (fptr, &dt->time_zone_offset) != 0))
181 {
182 return HH_FREAD_ERROR;
183 }
184
185 return 0;
186}

218int8_t
219read_file_flags (FILE *fptr, file_flags *ff)
220{
221 uint8_t byte;
222 if (read_single_uint8 (fptr, &byte) != 0)
223 return HH_FREAD_ERROR;
224
225 if (byte & 0x1)
226 {
227 ff->hidden = true;
228 }
229 if (byte & 0x2)
230 {
231 ff->subdirectory = true;
232 }
233 if (byte & 0x4)
234 {
235 ff->associated_file = true;
236 }
237 if (byte & 0x8)
238 {
239 ff->extended_attribute_record_contains_owner_and_group_permissions
240 = true;
241 }
242 if (byte & 0x10)
243 {
244 ff->final_directory_record = true;
245 }
246
247 return 0;
248}
