67void
68process_DAT_file (FILE *fptr)
69{
70 /*
71 * First 32k (up to 0x8000) stores the `system area` of the disk.
72 * Unused by HARVEST.DAT, HARVEST3.DAT, HARVEST4.DAT.
73 * See: https://wiki.osdev.org/ISO_9660#System_Area
74 */
75 fseek (fptr, 0x8000, SEEK_SET);
76
77 volume_descriptor vd;
78 process_volume_descriptor_header (fptr, &vd);
79
80 // Verify that this is a primary volume descriptor
81 if (vd.type_code != 0x01)
82 {
83 puts ("Error: Unexpected volume descriptor type code.");
84 printf ("\tExpected %02x, got %02x.\n", 0x01, vd.type_code);
85 puts ("Note: Extracting files from HARVEST2.DAT is currently "
86 "unsupported.");
87 fclose (fptr);
88 exit (1);
89 }
90
91 process_volume_descriptor_data (fptr, &vd.data);
92
93 size_t current_disk_name_length = strcspn (vd.data.volume_identifier, " ");
94 OP_CURRENT_DISK_NAME = vd.data.volume_identifier;
95 OP_CURRENT_DISK_NAME[current_disk_name_length] = '\0';
96
97 // TODO: print the volume descriptor header/data to some file/log.
98
99 // logical block size, in big endian form
100 const uint16_t LOGICAL_BLOCK_SIZE_BE
101 = change_endianness_uint16 (vd.data.logical_block_size);
102
103 // move to beginning of type-l path table
104 fseek (fptr, LOGICAL_BLOCK_SIZE_BE * vd.data.type_l_path_table_location,
105 SEEK_SET);
106
107 path_table pt;
108 if (create_path_table (&pt) != 0)
109 {
110 fclose (fptr);
111 exit (1);
112 }
113
114 process_type_l_path_table (fptr, &pt);
115
116 create_directories_and_extract_data_from_path_file (
117 fptr, LOGICAL_BLOCK_SIZE_BE, &pt);
118
119 // handle root directory
120 fseek (fptr, LOGICAL_BLOCK_SIZE_BE * pt.entries[0].location_of_extent,
121 SEEK_SET);
122
123 char *path = calloc (strlen (OUTPUT_DIR) + current_disk_name_length + 2,
124 sizeof (char));
125 if (path == NULL)
126 {
127 destroy_path_table (&pt);
128 return;
129 }
130
131 strcpy (path, OUTPUT_DIR);
132 strcat (path, "/");
133 strcat (path, OP_CURRENT_DISK_NAME);
134 extract_directory (fptr, LOGICAL_BLOCK_SIZE_BE, path);
135
136 free (path);
137 destroy_path_table (&pt);
138}
