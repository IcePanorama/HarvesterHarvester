81int8_t
82process_DAT_file (FILE *fptr)
83{
84 /*
85 * First 32k (up to 0x8000) stores the `system area` of the disk.
86 * Unused by HARVEST.DAT, HARVEST3.DAT, HARVEST4.DAT.
87 * See: https://wiki.osdev.org/ISO_9660#System_Area
88 */
89 fseek (fptr, 0x8000, SEEK_SET);
90
91 volume_descriptor vd;
92 if (process_volume_descriptor_header (fptr, &vd) != 0)
93 return -1;
94
95 // Verify that this is a primary volume descriptor
96 if (vd.type_code != 0x01)
97 {
98 puts ("Error: Unexpected volume descriptor type code.");
99 printf ("\tExpected %02x, got %02x.\n", 0x01, vd.type_code);
100 return -1;
101 }
102
103 if (process_volume_descriptor_data (fptr, &vd.data) != 0)
104 return -1;
105
106 size_t current_disk_name_length = strcspn (vd.data.volume_identifier, " ");
107 OP_CURRENT_DISK_NAME = vd.data.volume_identifier;
108 OP_CURRENT_DISK_NAME[current_disk_name_length] = '\0';
109
110 // TODO: print the volume descriptor header/data to some file/log.
111
112 // logical block size, in big endian form
113 const uint16_t block_size_be
114 = change_endianness_uint16 (vd.data.logical_block_size);
115
116 // move to beginning of type-l path table
117 fseek (fptr, block_size_be * vd.data.type_l_path_table_location, SEEK_SET);
118
119 path_table pt;
120 if (create_path_table (&pt) != 0)
121 return -1;
122
123 if (process_type_l_path_table (fptr, &pt) != 0)
124 {
125 destroy_path_table (&pt);
126 return -1;
127 }
128
129 if (create_directories_and_extract_data_from_path_file (fptr, block_size_be,
130 &pt)
131 != 0)
132 {
133 destroy_path_table (&pt);
134 return -1;
135 }
136
137 // handle root directory
138 fseek (fptr, block_size_be * pt.entries[0].location_of_extent, SEEK_SET);
139
140 char *path = calloc (strlen (OP_OUTPUT_DIR) + current_disk_name_length + 2,
141 sizeof (char));
142 if (path == NULL)
143 {
144 fprintf (stderr, CALLOC_FAILED_ERR_MSG_FMT,
145 strlen (OP_OUTPUT_DIR) + current_disk_name_length + 2);
146 destroy_path_table (&pt);
147 return -1;
148 }
149
150 strcpy (path, OP_OUTPUT_DIR);
151 strcat (path, "/");
152 strcat (path, OP_CURRENT_DISK_NAME);
153
154 if (extract_directory (fptr, block_size_be, path) != 0)
155 {
156 free (path);
157 destroy_path_table (&pt);
158 return -1;
159 }
160
161 free (path);
162 destroy_path_table (&pt);
163 return 0;
164}
