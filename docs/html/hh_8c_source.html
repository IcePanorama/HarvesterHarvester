35int
36main (int argc, char **argv)
37{
38 FILE *fptr = NULL;
39
40 if (argc >= 2)
41 handle_command_line_args (argc, argv);
42
43 if (OP_BATCH_PROCESS && !OP_SKIP_DAT_PROCESSING)
44 {
45 if (batch_process_DAT_files () != 0)
46 exit (1);
47 }
48 else if (!OP_SKIP_DAT_PROCESSING)
49 {
50 if (setup_extractor (&fptr, argv[argc - 1]) != 0)
51 exit (1);
52
53 if (process_DAT_file (fptr) != 0)
54 exit (1);
55
56 fclose (fptr);
57 }
58
59 /*
60 * TODO: Next steps:
61 * [] navigate to each output dir ("DISK1", "DISK2", etc.)
62 * [] Path is just `output/DISK#/`
63 * [] load the *.IDX file and use it to read the *.DAT files.
64 * [] extract files from those new *.DAT files into the directory they're
65 * already in.
66 */
67 if (process_new_dat_files () != 0)
68 {
69 exit (1);
70 }
71
72 return 0;
73}

94int8_t
95process_DAT_file (FILE *fptr)
96{
97 /*
98 * First 32k (up to 0x8000) stores the `system area` of the disk.
99 * Unused by HARVEST.DAT, HARVEST3.DAT, HARVEST4.DAT.
100 * See: https://wiki.osdev.org/ISO_9660#System_Area
101 */
102 fseek (fptr, 0x8000, SEEK_SET);
103
104 volume_descriptor vd;
105 if (process_volume_descriptor_header (fptr, &vd) != 0)
106 return -1;
107
108 // Verify that this is a primary volume descriptor
109 if (vd.type_code != 0x01)
110 {
111 puts ("Error: Unexpected volume descriptor type code.");
112 printf ("\tExpected %02x, got %02x.\n", 0x01, vd.type_code);
113 return -1;
114 }
115
116 if (process_volume_descriptor_data (fptr, &vd.data) != 0)
117 return -1;
118
119 size_t current_disk_name_length = strcspn (vd.data.volume_identifier, " ");
120 CURRENT_DISK_NAME = vd.data.volume_identifier;
121 CURRENT_DISK_NAME[current_disk_name_length] = '\0';
122
123 // TODO: print the volume descriptor header/data to some file/log.
124
125 // logical block size, in big endian form
126 const uint16_t block_size_be
127 = change_endianness_uint16 (vd.data.logical_block_size);
128
129 // move to beginning of type-l path table
130 fseek (fptr, block_size_be * vd.data.type_l_path_table_location, SEEK_SET);
131
132 path_table pt;
133 if (create_path_table (&pt) != 0)
134 return -1;
135
136 if (process_type_l_path_table (fptr, &pt) != 0)
137 {
138 destroy_path_table (&pt);
139 return -1;
140 }
141
142 if (create_directories_and_extract_data_from_path_file (fptr, block_size_be,
143 &pt)
144 != 0)
145 {
146 destroy_path_table (&pt);
147 return -1;
148 }
149
150 // handle root directory
151 fseek (fptr, block_size_be * pt.entries[0].location_of_extent, SEEK_SET);
152
153 char *path = calloc (strlen (OP_OUTPUT_DIR) + current_disk_name_length + 2,
154 sizeof (char));
155 if (path == NULL)
156 {
157 fprintf (stderr, CALLOC_FAILED_ERR_MSG_FMT,
158 strlen (OP_OUTPUT_DIR) + current_disk_name_length + 2);
159 destroy_path_table (&pt);
160 return -1;
161 }
162
163 strcpy (path, OP_OUTPUT_DIR);
164 strcat (path, "/");
165 strcat (path, CURRENT_DISK_NAME);
166
167 if (extract_directory (fptr, block_size_be, path) != 0)
168 {
169 free (path);
170 destroy_path_table (&pt);
171 return -1;
172 }
173
174 free (path);
175 destroy_path_table (&pt);
176 return 0;
177}

179int8_t
180batch_process_DAT_files ()
181{
182 const char *OPEN_INPUT_DIR_ERR_MSG_FMT
183 = "ERROR: Error opening input directory, %s.\n";
184 const uint8_t DAT_FILENAME_LEN = strlen ("HARVESTX.DAT");
185 char *filename;
186
187#ifdef _WIN32
188 WIN32_FIND_DATAA file_data;
189 HANDLE hFind;
190 char search_path[256];
191 strcpy (search_path, OP_INPUT_DIR);
192 strcat (search_path, "\\*");
193
194 hFind = FindFirstFileA (search_path, &file_data);
195 if (hFind == INVALID_HANDLE_VALUE)
196 {
197 fprintf (stderr, OPEN_INPUT_DIR_ERR_MSG_FMT, OP_INPUT_DIR);
198 return -1;
199 }
200
201 do
202 {
203 if (strcmp (file_data.cFileName, ".") == 0
204 || strcmp (file_data.cFileName, "..") == 0
205 || !is_string_dat_file (file_data.cFileName))
206 {
207 continue;
208 }
209
210 filename = calloc (strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN + 2,
211 sizeof (char));
212 if (filename == NULL)
213 {
214 fprintf (stderr, CALLOC_FAILED_ERR_MSG_FMT,
215 strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN + 2);
216 return HH_MEM_ALLOC_ERROR;
217 }
218
219 strcpy (filename, OP_INPUT_DIR);
220 strcat (filename, &OP_PATH_SEPARATOR);
221 strcat (filename, file_data.cFileName);
222
223 FILE *fptr = NULL;
224 if (setup_extractor (&fptr, filename) != 0)
225 {
226 free (filename);
227 FindClose (hFind);
228 return HH_FOPEN_ERROR;
229 }
230
231 if (process_DAT_file (fptr) != 0)
232 {
233 fclose (fptr);
234 free (filename);
235 FindClose (hFind);
236 return -1;
237 }
238
239 fclose (fptr);
240 free (filename);
241 }
242 while (FindNextFileA (hFind, &file_data) != 0);
243
244 FindClose (hFind);
245#else
246 struct dirent *entry;
247 DIR *dir;
248 dir = opendir (OP_INPUT_DIR);
249 if (dir == NULL)
250 {
251 fprintf (stderr, OPEN_INPUT_DIR_ERR_MSG_FMT, OP_INPUT_DIR);
252 return -1;
253 }
254
255 while ((entry = readdir (dir)) != NULL)
256 {
257 if (strcmp (entry->d_name, ".") == 0 || strcmp (entry->d_name, "..") == 0
258 || !is_string_dat_file (entry->d_name))
259 {
260 continue;
261 }
262
263 filename = calloc (strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN + 2,
264 sizeof (char));
265 if (filename == NULL)
266 {
267 fprintf (stderr, CALLOC_FAILED_ERR_MSG_FMT,
268 strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN + 2);
269 return HH_MEM_ALLOC_ERROR;
270 }
271
272 strcpy (filename, OP_INPUT_DIR);
273 strcat (filename, &OP_PATH_SEPARATOR);
274 strcat (filename, entry->d_name);
275
276 FILE *fptr = NULL;
277 if (setup_extractor (&fptr, filename) != 0)
278 {
279 free (filename);
280 closedir (dir);
281 return HH_FOPEN_ERROR;
282 }
283
284 if (process_DAT_file (fptr) != 0)
285 {
286 fclose (fptr);
287 free (filename);
288 closedir (dir);
289 return -1;
290 }
291
292 fclose (fptr);
293 free (filename);
294 }
295
296 closedir (dir);
297#endif
298 return 0;
299}

305int8_t
306process_new_dat_files (void)
307{
308 const uint32_t file_path_len = 3 + strlen (OP_OUTPUT_DIR) + strlen ("DISK#")
309 + strlen ("HARVEST2.DAT");
310 char *interior_dat_file_path = calloc (file_path_len, sizeof (char));
311 if (interior_dat_file_path == NULL)
312 {
313 fprintf (stderr, CALLOC_FAILED_ERR_MSG_FMT, file_path_len);
314 return HH_MEM_ALLOC_ERROR;
315 }
316
317 // for subdir in OP_OUTPUT_DIR
318 strcpy (interior_dat_file_path, OP_OUTPUT_DIR);
319 strcat (interior_dat_file_path, &OP_PATH_SEPARATOR);
320 strcat (interior_dat_file_path, "DISK1"); // replace w/ subdir name
321 strcat (interior_dat_file_path, &OP_PATH_SEPARATOR);
322 // probably need to figure out a good solution for storing these filenames
323 // for the files that we are expecting.
324 strcat (interior_dat_file_path, "INDEX.001");
325
326 FILE *fptr = NULL;
327 if (setup_extractor (&fptr, interior_dat_file_path) != 0)
328 {
329 free (interior_dat_file_path);
330 return HH_FOPEN_ERROR;
331 }
332
333 index_file idx_file;
334 if (create_index_file (&idx_file) != 0)
335 {
336 fclose (fptr);
337 free (interior_dat_file_path);
338 return -1;
339 }
340
341 if (process_index_file (fptr, &idx_file) != 0)
342 {
343 destroy_index_file (&idx_file);
344 fclose (fptr);
345 free (interior_dat_file_path);
346 return -1;
347 }
348
349 destroy_index_file (&idx_file);
350 fclose (fptr);
351 free (interior_dat_file_path);
352 return 0;
353}
