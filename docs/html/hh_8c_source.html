38int
39main (int argc, char **argv)
40{
41 FILE *fptr = NULL;
42
43 if (argc >= 2)
44 handle_command_line_args (argc, argv);
45
46 if (OP_BATCH_PROCESS && !OP_SKIP_DAT_PROCESSING)
47 {
48 if (batch_process_DAT_files () != 0)
49 exit (1);
50 }
51 else if (!OP_SKIP_DAT_PROCESSING)
52 {
53 if (setup_extractor (&fptr, argv[argc - 1]) != 0)
54 exit (1);
55
56 if (process_DAT_file (fptr) != 0)
57 exit (1);
58
59 fclose (fptr);
60 }
61
62 if (!OP_SKIP_INT_DAT_PROCESSING && process_internal_dat_files () != 0)
63 {
64 exit (1);
65 }
66
67 return 0;
68}

93int8_t
94process_DAT_file (FILE *fptr)
95{
96 /*
97 * First 32k (up to 0x8000) stores the `system area` of the disk.
98 * Unused by HARVEST.DAT, HARVEST3.DAT, HARVEST4.DAT.
99 * See: https://wiki.osdev.org/ISO_9660#System_Area
100 */
101 fseek (fptr, 0x8000, SEEK_SET);
102
103 volume_descriptor vd;
104 if (process_volume_descriptor_header (fptr, &vd) != 0)
105 return -1;
106
107 // Verify that this is a primary volume descriptor
108 if (vd.type_code != 0x01)
109 {
110 hh_log (HH_LOG_ERROR,
111 "Unexpected volume descriptor type code. Expected 0x%02X, got "
112 "0x%02X.",
113 0x01, vd.type_code);
114 return -1;
115 }
116
117 if (process_volume_descriptor_data (fptr, &vd.data) != 0)
118 return -1;
119
120 size_t current_disk_name_length = strcspn (vd.data.volume_identifier, " ");
121 CURRENT_DISK_NAME = vd.data.volume_identifier;
122 CURRENT_DISK_NAME[current_disk_name_length] = '\0';
123
124 // TODO: print the volume descriptor header/data to some file/log.
125
126 // logical block size, in big endian form
127 const uint16_t block_size_be
128 = change_endianness_uint16 (vd.data.logical_block_size);
129
130 // move to beginning of type-l path table
131 fseek (fptr, block_size_be * vd.data.type_l_path_table_location, SEEK_SET);
132
133 path_table pt;
134 if (create_path_table (&pt) != 0)
135 return -1;
136
137 if (process_type_l_path_table (fptr, &pt) != 0)
138 {
139 destroy_path_table (&pt);
140 return -1;
141 }
142
143 if (create_directories_and_extract_data_from_path_file (fptr, block_size_be,
144 &pt)
145 != 0)
146 {
147 destroy_path_table (&pt);
148 return -1;
149 }
150
151 // handle root directory
152 fseek (fptr, block_size_be * pt.entries[0].location_of_extent, SEEK_SET);
153
154 char *path = calloc (strlen (OP_OUTPUT_DIR) + current_disk_name_length + 2,
155 sizeof (char));
156 if (path == NULL)
157 {
158 destroy_path_table (&pt);
159 return handle_calloc_error (strlen (OP_OUTPUT_DIR)
160 + current_disk_name_length + 2);
161 }
162
163 strcpy (path, OP_OUTPUT_DIR);
164 strcat (path, &OP_PATH_SEPARATOR);
165 strcat (path, CURRENT_DISK_NAME);
166
167 if (extract_directory (fptr, block_size_be, path) != 0)
168 {
169 free (path);
170 destroy_path_table (&pt);
171 return -1;
172 }
173
174 free (path);
175 destroy_path_table (&pt);
176 return 0;
177}

179int8_t
180batch_process_DAT_files ()
181{
182 const char *OPEN_INPUT_DIR_ERR_MSG_FMT
183 = "[HarvesterHarvester]ERROR: Error opening input directory, %s.";
184 const uint8_t DAT_FILENAME_LEN = strlen ("HARVESTX.DAT");
185 char *filename;
186
187#ifdef _WIN32
188 WIN32_FIND_DATAA file_data;
189 HANDLE hFind;
190 char search_path[256];
191 strcpy (search_path, OP_INPUT_DIR);
192 strcat (search_path, "\\*");
193
194 hFind = FindFirstFileA (search_path, &file_data);
195 if (hFind == INVALID_HANDLE_VALUE)
196 {
197 hh_log (HH_LOG_ERROR, OPEN_INPUT_DIR_ERR_MSG_FMT, OP_INPUT_DIR);
198 return -1;
199 }
200
201 do
202 {
203 if (strcmp (file_data.cFileName, ".") == 0
204 || strcmp (file_data.cFileName, "..") == 0
205 || !is_string_dat_file (file_data.cFileName))
206 {
207 continue;
208 }
209
210 filename = calloc (strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN + 2,
211 sizeof (char));
212 if (filename == NULL)
213 {
214 return handle_calloc_error (strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN
215 + 2);
216 }
217
218 strcpy (filename, OP_INPUT_DIR);
219 strcat (filename, &OP_PATH_SEPARATOR);
220 strcat (filename, file_data.cFileName);
221
222 FILE *fptr = NULL;
223 if (setup_extractor (&fptr, filename) != 0)
224 {
225 free (filename);
226 FindClose (hFind);
227 return HH_FOPEN_ERROR;
228 }
229
230 if (process_DAT_file (fptr) != 0)
231 {
232 fclose (fptr);
233 free (filename);
234 FindClose (hFind);
235 return -1;
236 }
237
238 fclose (fptr);
239 free (filename);
240 }
241 while (FindNextFileA (hFind, &file_data) != 0);
242
243 FindClose (hFind);
244#else
245 struct dirent *entry;
246 DIR *dir;
247 dir = opendir (OP_INPUT_DIR);
248 if (dir == NULL)
249 {
250 hh_log (HH_LOG_ERROR, OPEN_INPUT_DIR_ERR_MSG_FMT, OP_INPUT_DIR);
251 return -1;
252 }
253
254 while ((entry = readdir (dir)) != NULL)
255 {
256 if (strcmp (entry->d_name, ".") == 0 || strcmp (entry->d_name, "..") == 0
257 || !is_string_dat_file (entry->d_name))
258 {
259 continue;
260 }
261
262 filename = calloc (strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN + 2,
263 sizeof (char));
264 if (filename == NULL)
265 {
266 return handle_calloc_error (strlen (OP_INPUT_DIR) + DAT_FILENAME_LEN
267 + 2);
268 }
269
270 strcpy (filename, OP_INPUT_DIR);
271 strcat (filename, &OP_PATH_SEPARATOR);
272 strcat (filename, entry->d_name);
273
274 FILE *fptr = NULL;
275 if (setup_extractor (&fptr, filename) != 0)
276 {
277 free (filename);
278 closedir (dir);
279 return HH_FOPEN_ERROR;
280 }
281
282 if (process_DAT_file (fptr) != 0)
283 {
284 fclose (fptr);
285 free (filename);
286 closedir (dir);
287 return -1;
288 }
289
290 fclose (fptr);
291 free (filename);
292 }
293
294 closedir (dir);
295#endif
296 return 0;
297}

299int8_t
300process_internal_dat_files (void)
301{
302 const char *interal_paths = "internal-dat-file-paths.txt";
303 FILE *table = fopen (interal_paths, "rb");
304 if (table == NULL)
305 {
306 return handle_fopen_error ((char *)interal_paths);
307 }
308
309 while (!peek_eof (table))
310 {
311 // TODO: convert as many malloc'd strings to char arrays as possible.
312 char index_file_path[256] = { 0 };
313 build_path_string_from_file (table, index_file_path);
314
315 FILE *fptr = NULL;
316 if (setup_extractor (&fptr, index_file_path) != 0)
317 {
318 hh_log (HH_LOG_WARNING, "Skipping...");
319
320 // Skip next line
321 build_path_string_from_file (table, index_file_path);
322
323 continue;
324 }
325
326 index_file idx_file;
327 if (create_index_file (&idx_file) != 0)
328 {
329 fclose (fptr);
330 fclose (table);
331 return -1;
332 }
333
334 hh_log (HH_LOG_INFO, "Processing index file: %s", index_file_path);
335 if (process_index_file (fptr, &idx_file) != 0)
336 {
337 destroy_index_file (&idx_file);
338 fclose (table);
339 fclose (fptr);
340 return -1;
341 }
342 fclose (fptr);
343
344 char dat_file_path[256] = { 0 };
345 build_path_string_from_file (table, dat_file_path);
346
347 if (extract_index_file (&idx_file, index_file_path, dat_file_path) != 0)
348 {
349 destroy_index_file (&idx_file);
350 return -1;
351 }
352
353 destroy_index_file (&idx_file);
354 }
355
356 fclose (table);
357 return 0;
358}
